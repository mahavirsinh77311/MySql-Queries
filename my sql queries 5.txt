**** select single query to get employee name, all hobby_name in single column ****

SELECT concat(employee.first_name,' ',  hobby.name) as employee_with_hobby
FROM employee
JOIN employee_hobby  ON employee_hobby.fk_employee_id  = employee.id
JOIN hobby ON employee_hobby.fk_hobby_id = hobby.id
order by employee_hobby.id


****  select query to get  employee name, his/her employee_salary ****

SELECT employee.first_name,  employee_salary.salary
FROM employee
JOIN employee_salary  ON employee_salary.fk_employee_id  = employee.id
order by employee_salary.id


****  select query to get employee name, total salary of employee, hobby name ****

SELECT CONCAT(e.first_name, ' ', e.last_name) AS full_name, 
      sum( es.salary) AS total_salary, 
       GROUP_CONCAT(h.name)
FROM employee_hobby AS eh 
LEFT JOIN hobby h ON h.id = eh.fk_hobby_id
INNER JOIN employee AS e ON e.id = eh.fk_employee_id    
INNER JOIN employee_salary AS es ON es.fk_employee_id = eh.fk_employee_id
GROUP BY eh.fk_employee_id;
